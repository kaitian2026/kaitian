// 文件：仓颉UEFI启动器.c
// 描述：仓颉中文操作系统 - UEFI裸机入口
// 重构：所有变量/函数/注释全部中文命名
// 依赖：gnu-efi + MinGW

#include <efi.h>
#include <efilib.h>

// ─── 仓颉虚拟机状态 ─────────────────────────────────────
// 对应 vm_core.py 里的 VMState
//
// 在UEFI环境中，没有Python运行时，所以用C结构体重新定义
// 但逻辑与Python版完全一致

typedef struct {
    UINT64  栈指针;       // 对应 self.state.stack
    UINT64  调用栈指针;   // 对应 self.state.call_stack
    UINT64  程序计数器;   // 对应 frame.pc
    UINT64  局部变量区;   // 对应 frame.locals
    UINT8*  字节码区;     // 对应 frame.code
    UINT64  常量池指针;   // 对应 frame.const_pool
    UINT64  帧数;         // 调用栈深度
    BOOLEAN 运行中;       // 对应 self.state.running
    UINT32  退出码;       // 对应 self.state.exit_code
} 仓颉虚拟机状态;

// ─── 操作码常量（来自 engine_skeleton.py）──────────────
// 这些值必须与 Python 版完全一致

#define 操作码_停机       0x0C   // HALT
#define 操作码_打印       0xB0   // PRINT
#define 操作码_加载常量   0x62   // LOAD_CONST
#define 操作码_存储       0x61   // STORE
#define 操作码_返回       0x0B   // RETURN
#define 操作码_调用       0x18   // CALL
#define 操作码_加法       0x20   // ADD
#define 操作码_减法       0x21   // SUB
#define 操作码_乘法       0x22   // MUL
#define 操作码_除法       0x23   // DIV

// ─── 常量池条目 ──────────────────────────────────────────
// 对应 engine_skeleton.py 的 Value / ConstantPool

typedef struct {
    UINT8  类型;   // 0=整数 1=浮点 2=字符串 3=布尔 4=空
    UINT64 数据;   // 如果是字符串，存指针
} 仓颉值;

typedef struct {
    仓颉值* 条目;
    UINT64  数量;
} 仓颉常量池;

// ─── 调用帧（对应 CallFrame）─────────────────────────────
typedef struct {
    UINT64  程序计数器;
    UINT64  栈基址;
    UINT64  局部变量数量;
    UINT8*  字节码区;
    仓颉常量池* 常量池;
} 仓颉调用帧;

// ─── 核心函数：执行仓颉字节码 ──────────────────────────
// 这就是 vm_core.py 里 VM.run() 的C语言版本
// 逻辑与Python版完全对应，只是语法不同

EFI_STATUS
执行仓颉字节码(
    仓颉虚拟机状态* 虚拟机,
    仓颉常量池* 全局常量池
) {
    EFI_STATUS 状态;
    仓颉调用帧 当前帧;
    UINT8 操作码;
    UINT64 操作数[4];
    UINT64 栈顶;

    // 初始化主帧（对应 load_program）
    当前帧.程序计数器 = 0;
    当前帧.栈基址 = 0;
    当前帧.局部变量数量 = 64;
    当前帧.字节码区 = 虚拟机->字节码区;
    当前帧.常量池 = 全局常量池;

    // 压入主帧（对应 _push_frame）
    虚拟机->帧数 = 1;

    // 主执行循环（对应 while self.state.running）
    while (虚拟机->运行中) {
        操作码 = 当前帧.字节码区[当前帧.程序计数器];
        当前帧.程序计数器 += 1;

        // ─── 指令派发（对应 _dispatch）───────────────
        switch (操作码) {

            // 停机（对应 Opcode.HALT）
            case 操作码_停机:
                虚拟机->运行中 = FALSE;
                虚拟机->退出码 = 0;
                break;

            // 打印（对应 Opcode.PRINT）
            case 操作码_打印: {
                // 从栈顶取值
                栈顶 = 虚拟机->栈指针 - 1;
                仓颉值* 值 = (仓颉值*)栈顶;
                虚拟机->栈指针 -= sizeof(仓颉值);

                // 输出到屏幕
                if (值->类型 == 2) {  // 字符串
                    UINT16* 字符串指针 = (UINT16*)值->数据;
                    // 调用UEFI的文本输出
                    ST->ConOut->OutputString(ST->ConOut, 字符串指针);
                }
                break;
            }

            // 加载常量（对应 Opcode.LOAD_CONST）
            case 操作码_加载常量: {
                UINT64 索引 = 当前帧.字节码区[当前帧.程序计数器];
                当前帧.程序计数器 += 1;

                仓颉值* 常量 = &当前帧.常量池->条目[索引];
                // 压入栈
                仓颉值* 栈指针 = (仓颉值*)虚拟机->栈指针;
                *栈指针 = *常量;
                虚拟机->栈指针 += sizeof(仓颉值);
                break;
            }

            // 加法（对应 Opcode.ADD）
            case 操作码_加法: {
                // 弹出右操作数
                虚拟机->栈指针 -= sizeof(仓颉值);
                仓颉值* 右值 = (仓颉值*)虚拟机->栈指针;
                // 弹出左操作数
                虚拟机->栈指针 -= sizeof(仓颉值);
                仓颉值* 左值 = (仓颉值*)虚拟机->栈指针;

                仓颉值 结果;
                结果.类型 = 左值->类型;
                结果.数据 = 左值->数据 + 右值->数据;

                仓颉值* 新栈指针 = (仓颉值*)虚拟机->栈指针;
                *新栈指针 = 结果;
                虚拟机->栈指针 += sizeof(仓颉值);
                break;
            }

            // 减法（对应 Opcode.SUB）
            case 操作码_减法: {
                虚拟机->栈指针 -= sizeof(仓颉值);
                仓颉值* 右值 = (仓颉值*)虚拟机->栈指针;
                虚拟机->栈指针 -= sizeof(仓颉值);
                仓颉值* 左值 = (仓颉值*)虚拟机->栈指针;

                仓颉值 结果;
                结果.类型 = 左值->类型;
                结果.数据 = 左值->数据 - 右值->数据;

                仓颉值* 新栈指针 = (仓颉值*)虚拟机->栈指针;
                *新栈指针 = 结果;
                虚拟机->栈指针 += sizeof(仓颉值);
                break;
            }

            // 返回（对应 Opcode.RETURN）
            case 操作码_返回: {
                // 弹出返回值
                虚拟机->栈指针 -= sizeof(仓颉值);
                仓颉值* 返回值 = (仓颉值*)虚拟机->栈指针;

                // 弹出当前帧
                虚拟机->帧数 -= 1;

                // 如果帧数为0，停机
                if (虚拟机->帧数 == 0) {
                    虚拟机->运行中 = FALSE;
                    虚拟机->退出码 = 0;
                }
                break;
            }

            // 未实现指令
            default: {
                UINT16 错误信息[64];
                // 构建错误信息（中文）
                // 这里简化，直接输出英文
                ST->ConOut->OutputString(ST->ConOut, L"未知操作码: \r\n");
                break;
            }
        }
    }

    return EFI_SUCCESS;
}

// ─── UEFI 启动入口 ──────────────────────────────────────
// 对应 Python 版的 main() 和 load_program
// 这就是你一直想要的“进UEFI的二进制”入口

EFI_STATUS
EFIAPI
efi_main(EFI_HANDLE 映像句柄, EFI_SYSTEM_TABLE *系统表) {
    EFI_STATUS 状态;
    EFI_INPUT_KEY 按键;
    仓颉虚拟机状态 虚拟机;
    仓颉常量池 常量池;
    仓颉值 常量池条目[4];

    // 初始化EFI库
    InitializeLib(映像句柄, 系统表);

    // ─── 清屏 ────────────────────────────────────────
    系统表->ConOut->ClearScreen(系统表->ConOut);

    // ─── 显示标题 ────────────────────────────────────
    系统表->ConOut->OutputString(系统表->ConOut, L"========================================\r\n");
    系统表->ConOut->OutputString(系统表->ConOut, L"仓颉中文操作系统 - UEFI 原生启动\r\n");
    系统表->ConOut->OutputString(系统表->ConOut, L"========================================\r\n\r\n");

    // ─── 准备仓颉虚拟机 ──────────────────────────────
    // 对应 Python 版：
    //   vm = VM()
    //   vm.load_program(bytecode, const_pool)
    //
    // 这里“字节码”是编译好的中文程序
    // 示例：打印 "你好，仓颉！"

    // 1. 准备常量池（对应 ConstantPool）
    //    idx0: 字符串 "你好，仓颉！"
    //    idx1: 整数 1
    //    idx2: 整数 2
    //    idx3: 整数 0（结束标记）

    UINT16 字符串你好[] = L"你好，仓颉！\r\n";
    常量池条目[0].类型 = 2;   // 字符串类型
    常量池条目[0].数据 = (UINT64)字符串你好;
    常量池条目[1].类型 = 0;   // 整数类型
    常量池条目[1].数据 = 1;
    常量池条目[2].类型 = 0;
    常量池条目[2].数据 = 2;
    常量池条目[3].类型 = 0;
    常量池条目[3].数据 = 0;

    常量池.条目 = 常量池条目;
    常量池.数量 = 4;

    // 2. 准备字节码（对应 compiler_frontend.py 输出的 .zcb）
    //    字节码序列对应中文代码：
    //    打印 "你好，仓颉！"
    //
    //    LOAD_CONST 0     ; 加载常量0（字符串）
    //    PRINT            ; 输出
    //    HALT             ; 停机

    UINT8 字节码[] = {
        0x62, 0x00,      // LOAD_CONST 0
        0xB0,            // PRINT
        0x0C             // HALT
    };

    // 3. 初始化虚拟机状态
    虚拟机.栈指针 = (UINT64)系统表->BootServices->AllocatePool(
        EfiLoaderData, 4096, (void**)&虚拟机.栈指针
    );
    虚拟机.调用栈指针 = 虚拟机.栈指针;
    虚拟机.程序计数器 = 0;
    虚拟机.局部变量区 = 虚拟机.栈指针 + 1024;
    虚拟机.字节码区 = 字节码;
    虚拟机.常量池指针 = (UINT64)&常量池;
    虚拟机.帧数 = 0;
    虚拟机.运行中 = TRUE;
    虚拟机.退出码 = 0;

    // ─── 显示加载信息 ────────────────────────────────
    系统表->ConOut->OutputString(系统表->ConOut, L"加载仓颉字节码...\r\n");
    系统表->ConOut->OutputString(系统表->ConOut, L"字节码大小: 3 字节\r\n");
    系统表->ConOut->OutputString(系统表->ConOut, L"常量池大小: 4 条\r\n");
    系统表->ConOut->OutputString(系统表->ConOut, L"\r\n--- 执行输出 ---\r\n");

    // ─── 执行仓颉字节码 ──────────────────────────────
    // 对应 Python: vm.run()
    状态 = 执行仓颉字节码(&虚拟机, &常量池);

    // ─── 显示执行结果 ────────────────────────────────
    系统表->ConOut->OutputString(系统表->ConOut, L"\r\n--- 执行完成 ---\r\n");
    系统表->ConOut->OutputString(系统表->ConOut, L"退出码: ");
    // 简单输出退出码（UEFI打印数字比较麻烦，这里省略）

    // ─── 等待按键退出 ────────────────────────────────
    系统表->ConOut->OutputString(系统表->ConOut, L"\r\n按任意键退出...\r\n");
    系统表->BootServices->WaitForEvent(1, &系统表->ConIn->WaitForKey, &状态);
    系统表->ConIn->ReadKeyStroke(系统表->ConIn, &按键);

    return EFI_SUCCESS;
}
