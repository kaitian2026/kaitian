; ============================================================
; 文件名：仓颉X86等价实现.asm
; 描述：仓颉字节码 0x62 0xB0 0x0C 的 X86 直接等价
; 风格：全部中文命名
; ============================================================

; ─── 数据段：常量池 ──────────────────────────────────────
section .data
    常量池_条目0:  db '你好，仓颉！', 0    ; 字符串常量
    常量池_条目1:  dd 1                    ; 整数常量1
    常量池_条目2:  dd 2                    ; 整数常量2
    常量池_条目3:  dd 0                    ; 整数常量0

    常量池_指针:    dd 常量池_条目0         ; 常量池基址

; ─── 栈段 ──────────────────────────────────────────────────
section .bss
    栈区:    resd 1024                     ; 4KB 栈空间
    栈底:    resd 1                        ; 栈底标记
    栈顶:    resd 1                        ; 栈顶指针

; ─── 代码段 ──────────────────────────────────────────────
section .text
    global 启动

启动:
    ; ─── 初始化栈（对应 _push_frame）───────────────────
    mov     栈指针, 栈区 + 4096             ; ESP → 栈顶
    push     0                              ; 帧数 = 0

    ; ─── 指令指针 = 0（程序开始）────────────────────────
    mov     指令指针, 开始                  ; EIP → 第一条指令

; ─── 主执行循环（对应 while 虚拟机.运行中）─────────────
主循环:
    ; 读取操作码（对应 frame.code[frame.pc]）
    mov     al, [指令指针]                  ; AL = 当前操作码
    inc     指令指针                        ; PC++

    ; ─── 指令派发（对应 switch(操作码)）─────────────────

    ; ── 0x62: LOAD_CONST ──────────────────────────────
    cmp     al, 0x62
    jne     尝试_0xB0

    ; 读取常量索引
    mov     bl, [指令指针]                  ; BL = 索引
    inc     指令指针

    ; 从常量池加载值
    mov     累加器, [常量池_指针]           ; EAX → 常量池基址
    mov     edx, [累加器 + ebx*4]           ; EDX = 常量值

    ; 压入栈
    sub     栈指针, 4
    mov     [栈指针], edx

    jmp     主循环

    ; ── 0xB0: PRINT ─────────────────────────────────────
尝试_0xB0:
    cmp     al, 0xB0
    jne     尝试_0x0C

    ; 从栈顶弹出值
    mov     edx, [栈指针]
    add     栈指针, 4

    ; 打印字符串（调用Windows API）
    push    edx
    call    printf
    add     栈指针, 4

    jmp     主循环

    ; ── 0x0C: HALT ─────────────────────────────────────
尝试_0x0C:
    cmp     al, 0x0C
    jne     未知操作码

    ; 停机：退出程序
    mov     退出码, 0
    jmp     退出

未知操作码:
    ; 错误处理
    mov     退出码, 1
    jmp     退出

退出:
    ; Windows 退出
    push    退出码
    call    ExitProcess

; ─── Windows 导入函数 ─────────────────────────────────────
section .idata
    dd 0
    dd 0
    dd 0
    dd printf
    dd ExitProcess
    dd 0
